# 宏

预处理（预编译）指令

## 预处理

    nasm -E %< -o %@

## 条件汇编

```s
%ifndef MACRO_INC
%define MACRO_INC

extern printf
extern exit

%endif
```

## 宏函数

```s
%include "macro.inc"

%define square(x) (x * x)
%define add(x, y) (x + y)

section .text
global main
main:
    mov eax, square(10)
    mov ebx, add(5, 3)
    ret
```

## 多行宏

```s
; 参数：
; - 基地址
; - 界限
; - type
; - segment
; - dpl
; - present
; - long_mode
; - big
; - granularity
%macro descriptor 9
    dw %2 & 0xffff; limit_low
    dw %1 & 0xffff; base_low_16
    db (%1 >> 16) & 0xff; base_low_8
    db %3 | (%4 << 4) | (%5 << 6) | (%6 << 7)
    db ((%2 >> 16) & 0xf) | (%7 << 5) | (%8 << 6) | (%9 << 7)
    db (%1 >> 24) & 0xff
%endmacro

descriptor 0x10000, 0xfff, 0b0010, 1, 0, 1, 0, 1, 0
```
